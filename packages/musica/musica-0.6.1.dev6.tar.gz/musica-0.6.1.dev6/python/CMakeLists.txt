  PUBLIC  
    $<BUILD_INTERFACE:${CMAKE_SOURCE_DIR}/lib/micm/include>
    $<INSTALL_INTERFACE:${MUSICA_INSTALL_INCLUDE_DIR}>
)

target_link_libraries(musica_python PUBLIC musica::musica)

install(TARGETS musica_python musica
  LIBRARY DESTINATION .
)
